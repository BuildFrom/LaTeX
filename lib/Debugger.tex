    % DEBUGGER: Color margins only
    % Limitation: Margins OR Body OR ROWS, not more than one
    \backgroundsetup{
        scale=1,
        angle=0,
        opacity=1,
        color=red!20,
        contents={
            \begin{tikzpicture}[remember picture,overlay]
                \fill (current page.north west) rectangle ([xshift=0.75in,yshift=-0.75in]current page.south west);
                \fill (current page.north east) rectangle ([xshift=-0.75in,yshift=-0.75in]current page.south east);
                \fill (current page.north west) rectangle ([yshift=-1in]current page.north east);
                \fill (current page.south west) rectangle ([yshift=1in]current page.south east);
            \end{tikzpicture}
        }
    }
\end{comment}
